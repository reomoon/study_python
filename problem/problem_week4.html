# 1) 재입력받아 소수 판별
def input_int(prompt='숫자 입력: '):
  while True:
    s = input(prompt)
    try:
      return int(s)
    except ValueError:
      print('입력 오류: 정수를 입력하세요.')

def is_prime(n):
  if n < 2:
    return False
  for i in range(2, int(n**0.5)+1):
    if n % i == 0:
      return False
  return True

num = input_int('정수 하나를 입력하세요: ')
print(f'{num}은 소수입니다' if is_prime(num) else f'{num}은 소수가 아닙니다')

# 2) 리스트 평균 계산 (예외 처리 포함)
def parse_number_list(s):
  parts = s.split()
  nums = []
  for p in parts:
    try:
      nums.append(float(p))
    except ValueError:
      print(f'경고: {p}는 숫자가 아니므로 무시합니다')
  return nums

s = input('숫자들을 공백으로 구분해 입력하세요: ')
nums = parse_number_list(s)
if nums:
  print('평균:', sum(nums)/len(nums))
else:
  print('유효한 숫자가 없습니다')

# 3) 커스텀 예외 사용 예
class MyError(Exception):
  pass

def check_positive(n):
  if n < 0:
    raise MyError('음수는 허용되지 않습니다')

try:
  x = int(input('음이 아닌 정수를 입력하세요: '))
  check_positive(x)
  print('입력 OK')
except MyError as e:
  print('처리 오류:', e)
except ValueError:
  print('정수로 변환할 수 없습니다')
